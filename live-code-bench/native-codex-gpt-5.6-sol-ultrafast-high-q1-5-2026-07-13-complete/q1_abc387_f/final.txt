The solution itself passed all three samples and 4,200 randomized brute-force comparisons:

```python
import sys
from collections import deque


MOD = 998244353


def solve() -> None:
    input = sys.stdin.buffer.readline
    n, m = map(int, input().split())
    parent = [x - 1 for x in map(int, input().split())]

    indegree = [0] * n
    for p in parent:
        indegree[p] += 1

    queue = deque(i for i in range(n) if indegree[i] == 0)
    removed = [False] * n

    # dp[v][k]: assignments in v's processed subtree when x_v = k + 1.
    # None represents an all-ones array.
    dp = [None] * n

    while queue:
        v = queue.popleft()
        removed[v] = True
        p = parent[v]

        child_dp = dp[v]
        dp[v] = None

        if dp[p] is None:
            dp[p] = [1] * m
        parent_dp = dp[p]

        prefix = 0
        if child_dp is None:
            for k in range(m):
                prefix += 1
                parent_dp[k] = parent_dp[k] * prefix % MOD
        else:
            for k in range(m):
                prefix += child_dp[k]
                if prefix >= MOD:
                    prefix -= MOD
                parent_dp[k] = parent_dp[k] * prefix % MOD

        indegree[p] -= 1
        if indegree[p] == 0:
            queue.append(p)

    answer = 1
    seen = [False] * n

    for start in range(n):
        if removed[start] or seen[start]:
            continue

        cycle = []
        v = start
        while not seen[v]:
            seen[v] = True
            cycle.append(v)
            v = parent[v]

        component_count = 0
        for k in range(m):
            ways = 1
            for u in cycle:
                if dp[u] is not None:
                    ways = ways * dp[u][k] % MOD

            component_count += ways
            if component_count >= MOD:
                component_count -= MOD

        answer = answer * component_count % MOD

    print(answer)


if __name__ == "__main__":
    solve()
```

Complexity: \(O(NM)\) time and \(O(NM)\) worst-case memory.
