```python
import sys


def solve() -> None:
    input = sys.stdin.readline
    N, M, A, B = map(int, input().split())
    intervals = [tuple(map(int, input().split())) for _ in range(M)]

    full_mask = (1 << B) - 1

    # State bit k indicates whether the square k positions before
    # the current square is reachable.
    #
    # For a good next square:
    #   new_dp = OR(state[A-1:B])
    # and all previous values shift one position back.
    transition = [0] * B
    transition[0] = full_mask ^ ((1 << (A - 1)) - 1)
    for i in range(1, B):
        transition[i] = 1 << (i - 1)

    # Boolean-matrix powers of the transition.
    powers = [transition]
    for _ in range(1, N.bit_length()):
        previous = powers[-1]
        squared = []

        for row in previous:
            combined = 0
            bits = row
            while bits:
                lowest = bits & -bits
                index = lowest.bit_length() - 1
                combined |= previous[index]
                bits -= lowest
            squared.append(combined)

        powers.append(squared)

    def advance_good(state: int, length: int) -> int:
        """Advance through `length` consecutive good squares."""
        power_index = 0

        while length:
            if length & 1:
                rows = powers[power_index]
                new_state = 0

                for i, row in enumerate(rows):
                    if state & row:
                        new_state |= 1 << i

                state = new_state
                if state == 0:
                    return 0

            length >>= 1
            power_index += 1

        return state

    # Square 1 is reachable and is guaranteed not to be bad.
    state = 1
    current = 1

    for left, right in intervals:
        # Process the good squares before this bad interval.
        state = advance_good(state, left - current - 1)
        if state == 0:
            print("No")
            return

        # Every bad square contributes an unreachable value (zero).
        bad_length = right - left + 1
        if bad_length >= B:
            print("No")
            return

        state = (state << bad_length) & full_mask
        if state == 0:
            print("No")
            return

        current = right

    # Process all remaining good squares, including square N.
    state = advance_good(state, N - current)

    print("Yes" if state & 1 else "No")


if __name__ == "__main__":
    solve()
```
