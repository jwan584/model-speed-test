```python
import sys


def main():
    input = sys.stdin.readline
    n = int(input())
    a = list(map(int, input().split()))

    top = 0
    bottom = (n + 1) // 2
    pairs = 0

    while top < n // 2 and bottom < n:
        if 2 * a[top] <= a[bottom]:
            pairs += 1
            top += 1
            bottom += 1
        else:
            bottom += 1

    print(pairs)


if __name__ == "__main__":
    main()
```
